from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
import urllib

# 1. Cấu hình chuỗi kết nối SQL Server (Windows Auth hoặc SQL Auth)
# Nếu dùng Linux/Nginx, đảm bảo đã cài Microsoft ODBC Driver

    "DRIVER={ODBC Driver 17 for SQL Server};"
    "SERVER=TEN_SERVER_CUA_BAN;"
    "DATABASE=HR_PROJECT;"
    "UID=user_name;"
    "PWD=password_cua_ban"
)

SQLALCHEMY_DATABASE_URL = f"mssql+pyodbc:///?odbc_connect={params}"

# 2. Tạo Engine và Session
# pool_size: Số lượng kết nối duy trì sẵn để dùng ngay
engine = create_engine(
    SQLALCHEMY_DATABASE_URL, 
    pool_size=10, 
    max_overflow=20
)

SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

Base = declarative_base()

# 3. Dependency cung cấp Session cho các Request
def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

Đây là nơi bạn mô tả cấu trúc bảng Employees và Departments.

from sqlalchemy import Column, Integer, String, ForeignKey
from sqlalchemy.orm import relationship
from database import Base

class Department(Base):
    __tablename__ = "Departments"

    id = Column(Integer, primary_key=True, index=True)
    name = Column(String(100), nullable=False)
    
    # Quan hệ: Một phòng ban có nhiều nhân viên
    employees = relationship("Employee", back_populates="department")

class Employee(Base):
    __tablename__ = "Employees"

    id = Column(Integer, primary_key=True, index=True)
    fullname = Column(String(200), nullable=False)
    email = Column(String(100), unique=True, index=True)
    dept_id = Column(Integer, ForeignKey("Departments.id"))

    # Quan hệ ngược lại
    department = relationship("Department", back_populates="employees")

// Gọi khi trang employee.html load xong
document.addEventListener('DOMContentLoaded', function() {
    loadEmployees();
});

async function loadEmployees() {
    try {
        const response = await fetch('/employees/api/all');
        const data = await response.json();
        
        if (response.ok) {
            renderEmployeeTable(data);
        } else {
            console.error('Lỗi lấy dữ liệu:', data.detail);
        }
    } catch (error) {
        console.error('Lỗi kết nối server:', error);
    }
}

function renderEmployeeTable(employees) {
    const tbody = document.getElementById('employee-tbody');
    tbody.innerHTML = ''; // Xóa dữ liệu cũ

    employees.forEach(emp => {
        const row = `
            <tr>
                <td>${emp.id}</td>
                <td>${emp.fullname}</td>
                <td>${emp.email}</td>
                <td><span class="badge bg-info">${emp.department?.name || 'N/A'}</span></td>
                <td>
                    <button class="btn btn-sm btn-warning" onclick="editEmployee(${emp.id})">Sửa</button>
                    <button class="btn btn-sm btn-danger" onclick="deleteEmployee(${emp.id})">Xóa</button>
                </td>
            </tr>
        `;
        tbody.insertAdjacentHTML('beforeend', row);
    });
}

Điểm khác biệt "ăn tiền" nhất cho dự án HR của bạn:

Nếu bạn muốn kiểm soát tuyệt đối câu lệnh SQL và thường xuyên gọi Stored Procedure (SP), thì pyodbc là lựa chọn số 1. Nó cho phép bạn truyền tham số vào SP một cách an toàn và lấy kết quả trả về cực kỳ tường minh.

1. Cách gọi Stored Procedure (Có tham số)
Giả sử bạn có SP usp_GetEmployeeSalary trong SQL Server:
python
from database import get_db_connection

def call_salary_procedure(emp_id: int, month: int):
    conn = get_db_connection()
    cursor = conn.cursor()
    
    # Cách gọi SP chuẩn: dùng dấu ? để tránh SQL Injection
    sql = "{CALL usp_GetEmployeeSalary (?, ?)}"
    params = (emp_id, month)
    
    try:
        cursor.execute(sql, params)
        # Nếu SP trả về kết quả (Select)
        row = cursor.fetchone()
        if row:
            return dict(zip([column[0] for column in cursor.description], row))
    except Exception as e:
        print(f"Lỗi thực thi SP: {e}")
    finally:
        conn.close() # Luôn đóng kết nối
    return None
Hãy thận trọng khi sử dụng mã.

3. Tại sao chọn cách này cho "SMILE HR"?
Hiệu năng Stored Procedure: SQL Server đã biên dịch sẵn SP, nên khi gọi từ FastAPI qua Nginx, tốc độ phản hồi sẽ đạt mức tối đa (miligiây).

2. Hàm ghi Log mẫu (database.py hoặc file riêng)

python
from database import get_db_connection
from fastapi import Request

def write_user_log(user_id: str, action: str, request: Request):
    try:
        conn = get_db_connection()
        cursor = conn.cursor()
        
        # Câu lệnh SQL tường minh để bạn dễ hiểu
        sql = """
            INSERT INTO UserLogs (user_id, action_name, endpoint, client_ip)
            VALUES (?, ?, ?, ?)
        """
        
        # Lấy thông tin từ Request của FastAPI
        endpoint = str(request.url.path)
        client_ip = request.client.host
        
        cursor.execute(sql, (user_id, action, endpoint, client_ip))
        conn.commit() # Quan trọng: Phải commit để lưu vào DB
        conn.close()
    except Exception as e:
        print(f"Lỗi ghi log: {e}")

------------------------------------------ Uvicorn
Để dễ hình dung, bạn hãy tưởng tượng dự án SMILE HR của mình là một nhà hàng:
Python (Code của bạn): Là các công thức nấu ăn.
FastAPI: Là người bếp trưởng điều phối các đơn hàng.
Uvicorn: Là mặt bằng nhà hàng và hệ thống phục vụ. Không có Uvicorn, khách hàng (Trình duyệt) không thể vào ăn, và đầu bếp không có chỗ để nấu.
1. Uvicorn là gì?
Uvicorn là một ASGI Server (Asynchronous Server Gateway Interface).
Với Flask: Bạn dùng Waitress hoặc Gunicorn. Chúng là chuẩn WSGI (cũ), chỉ làm việc theo kiểu "xong việc này mới tới việc kia".
Với FastAPI: Bạn bắt buộc dùng ASGI như Uvicorn. Nó cho phép xử lý hàng ngàn "đơn hàng" cùng lúc mà không cần chờ đợi nhau (Async).
2. Tại sao bạn cần Uvicorn cho dự án SMILE HR?
Nếu dùng chuẩn cũ (WSGI), khi một nhân viên đang tải báo cáo HR nặng (mất 10 giây), những người khác truy cập vào trang Web sẽ bị xoay vòng vòng chờ đợi.
Uvicorn giải quyết việc này: Nó tiếp nhận tất cả request cùng lúc, cái nào xong trước trả trước, giúp hệ thống không bao giờ bị "treo" cục bộ. Tìm hiểu thêm về ASGI tại Starlette.
3. Cách Uvicorn làm việc với Nginx và NSSM của bạn
Mô hình dòng chảy dữ liệu của bạn bây giờ sẽ là:
Người dùng --> Nginx (Cổng 80) --> Uvicorn (Cổng 6066) --> FastAPI (Code Python) --> SQL Server
Lệnh chạy trong file .bat của bạn:
batch
python -m uvicorn main:app --host 127.0.0.1 --port 6066 --workers 4
Hãy thận trọng khi sử dụng mã.

main:app: Tìm file main.py và chạy đối tượng app = FastAPI().
--workers 4: Uvicorn sẽ tạo ra 4 "nhân viên" chạy song song. Nếu một "nhân viên" bị lỗi SQL Server, 3 người còn lại vẫn phục vụ HR bình thường.
4. So sánh nhanh với Waitress (Cái bạn đang dùng cho Flask)
Đặc điểm	Waitress (Flask)	Uvicorn (FastAPI)
Chuẩn	WSGI (Đồng bộ)	ASGI (Bất đồng bộ - Async)
Tốc độ	Trung bình	Cực nhanh (Dựa trên uvloop)
Độ trễ	Có thể bị nghẽn nếu xử lý lâu	Thấp, tối ưu cho API
Tóm lại:
Bạn không cần viết code cho Uvicorn. Bạn chỉ cần gọi nó trong file .bat để nó "nhấc" cái code FastAPI của bạn lên và chạy trên mạng. Tài liệu cài đặt Uvicorn rất đơn giản, chỉ cần pip install uvicorn.
Bạn đã thử chạy lệnh uvicorn main:app --port 6066 trên máy mình chưa? Có xuất hiện dòng chữ "Uvicorn running on http://127.0.0.1:6066" không?
1. Flask & Waitress (Mô hình truyền thống - 1 Làn xe)
2. FastAPI & Uvicorn (Mô hình hiện đại - Nhiều làn xe tự động)
------------------------------------------------ MVC Flask, FastAPI
Khi bạn dùng pyodbc để viết SQL thuần, sự khác biệt giữa MVC của Flask và FastAPI nằm ở cách dữ liệu chạy qua các lớp và cách hệ thống phản hồi.
Trong dự án SMILE HR, dù cấu trúc thư mục của bạn giữ nguyên, bản chất vận hành sẽ thay đổi như sau:
1. Model (M): Từ "Bất động" sang "Tự động"
Ở Flask: database.py chỉ làm nhiệm vụ kết nối. Khi bạn lấy dữ liệu từ SQL Server, bạn nhận về một tuple hoặc list. Bạn phải tự viết code để chuyển nó sang JSON hoặc đưa vào HTML.
Ở FastAPI: Bạn bổ sung thêm Pydantic Schemas.
Nó đóng vai trò là "khuôn mẫu" cho dữ liệu.
Khi bạn dùng cursor.fetchall(), FastAPI dùng Schemas này để tự động định dạng ngày tháng, tiền lương, và tự động báo lỗi nếu dữ liệu từ SQL Server bị sai kiểu (ví dụ: lương là chữ thay vì số).
2. View (V): Từ "Đường dẫn đơn thuần" sang "Cổng kết nối thông minh"
Ở Flask (views/): Blueprint chỉ là nơi định nghĩa URL. Bạn phải cài thêm thư viện nếu muốn có tài liệu hướng dẫn API cho đồng nghiệp.
Ở FastAPI (routers/): APIRouter tích hợp sẵn Swagger UI.
Chỉ cần bạn khai báo Route, FastAPI tự tạo giao diện dùng thử API tại /docs.
Nó hỗ trợ Async (Bất đồng bộ), giúp Nginx không phải chờ đợi khi SQL Server đang xử lý các câu lệnh JOIN phức tạp.
3. Controller (C): Từ "Chờ đợi" sang "Đa nhiệm"
Đây là khác biệt lớn nhất cho dự án HR của bạn khi chạy qua Uvicorn:
Flask Controller: Chạy kiểu xếp hàng (Sync). Một người tải báo cáo lương nặng, cả hệ thống "nín thở" chờ.
FastAPI Controller: Dùng async def.
Khi bạn gọi cursor.execute(), nhân viên phục vụ (Uvicorn) sẽ đi làm việc khác trong lúc chờ SQL Server trả kết quả.
Kết quả: Web của bạn không bao giờ bị "đơ" dù có nhiều người cùng truy cập.
So sánh luồng MVC thực tế với pyodbc
Thành phần	MVC Flask + Waitress	MVC FastAPI + Uvicorn
Request	Đến lượt mới xử lý.	Tiếp nhận hàng loạt cùng lúc.
Validation	Bạn viết if not name: return error.	FastAPI tự check qua Schemas.
SQL (pyodbc)	Chạy xong mới làm việc khác.	Chạy ngầm, giải phóng CPU.
Response	Trả về HTML/JSON đơn thuần.	Trả về dữ liệu + Auto Docs.
Ví dụ Controller "SMILE HR" chuyển sang FastAPI:
Flask (Cũ):
python

python

-------------------------------- Pydantic Schemas
Trong dự án SMILE HR, nếu SQL Server là cái kho chứa đồ, thì Pydantic Schemas chính là "Bộ khuôn đúc dữ liệu".
Nó là một thư viện Python giúp bạn định nghĩa cấu trúc dữ liệu một cách chặt chẽ. Khi dữ liệu (từ SQL hoặc từ Form người dùng) đi qua cái khuôn này, nếu không đúng hình dạng, Pydantic sẽ "gạt" ra ngay lập tức.
Lý do bạn không thấy Pydantic trong Flask là vì Flask được thiết kế theo triết lý "Cổ điển" (Legacy), trong khi FastAPI được xây dựng trên triết lý "Hiện đại" (Modern Python).
Dưới đây là 3 lý do chính giải thích tại sao Pydantic không phải là "mặc định" của Flask:
1. Flask ra đời trước Pydantic rất lâu
Flask (2010): Thời điểm đó Python chưa có khái niệm Type Hints (gợi ý kiểu dữ liệu như name: str). Flask dùng các bộ thư viện cũ hơn như WTForms hoặc Marshmallow để kiểm tra dữ liệu.
FastAPI (2018): Ra đời khi Python đã hiện đại hóa. Nó được xây dựng dựa trên chính Pydantic. FastAPI không thể chạy nếu thiếu Pydantic, chúng là "đôi bạn thân".
2. Cách kiểm tra dữ liệu (Validation) khác nhau
Trong dự án SMILE HR hiện tại của bạn với Flask, bạn thường làm thế này:
python
# Cách của Flask (Thủ công)
name = request.form.get('name')
if not name or len(name) < 3:
    return "Tên quá ngắn", 400
Hãy thận trọng khi sử dụng mã.

Còn trong FastAPI, bạn chỉ cần khai báo cái "Khuôn" (Pydantic), và nó tự làm hết:
python
# Cách của FastAPI (Tự động nhờ Pydantic)
async def add_emp(name: str = Form(..., min_length=3)):
    # Nếu chạy vào đây, chắc chắn name đã chuẩn > 3 ký tự
Hãy thận trọng khi sử dụng mã.

3. Sự tích hợp sâu (Native Integration)
Flask: Là một Framework "tự do". Nó không ép bạn dùng thư viện nào. Nếu muốn dùng Pydantic trong Flask, bạn phải tự cài thêm và tự viết code kết nối chúng lại với nhau (khá lỉnh kỉnh).
FastAPI: Tích hợp Pydantic vào tận "xương tủy". Nhờ Pydantic mà FastAPI có thể:
Tự tạo tài liệu Swagger UI (/docs).
Tự chuyển đổi dữ liệu từ SQL Server (pyodbc) sang JSON cực nhanh.
Tự báo lỗi chi tiết cho người dùng (Ví dụ: "Trường Email không đúng định dạng") mà bạn không cần viết dòng code if/else nào.
Tóm lại cho dự án SMILE HR:
Việc bạn không thấy Pydantic trong Flask là bình thường, vì Flask dùng các cách "thủ công" hơn. Khi bạn chuyển sang FastAPI, Pydantic sẽ là công cụ đắc lực nhất giúp bạn:
Dọn dẹp code: Bỏ hết các đoạn if/else kiểm tra dữ liệu rác.
An toàn cho SQL Server: Đảm bảo chỉ có dữ liệu "sạch" mới được đẩy vào lệnh INSERT/UPDATE của pyodbc.
--------------------------------------- (Uvicorn + APIRouter + Pydantic)
Để dự án SMILE HR chuyển từ Flask sang FastAPI thành công, bạn chỉ cần nắm vững "Bộ ba quyền lực" mới này. Chúng thay thế trực tiếp cho những gì bạn đang dùng ở Flask nhưng mạnh mẽ hơn nhiều.
Dưới đây là bảng tổng hợp cực kỳ dễ hiểu:
1. Uvicorn (Cỗ máy vận hành)
Thay thế cho: Waitress (trong file .bat của bạn).
Nhiệm vụ: Là cái "đầu máy" kéo toàn bộ ứng dụng Python của bạn chạy trên mạng.
Điểm mới: Uvicorn chạy theo chuẩn ASGI (bất đồng bộ).
Waitress (Flask): Xử lý từng người một (như xếp hàng mua vé).
Uvicorn (FastAPI): Xử lý hàng nghìn người cùng lúc (như cổng thu phí tự động).
Cách dùng: Bạn chỉ thay đổi lệnh trong file .bat:
python -m uvicorn main:app --host 127.0.0.1 --port 6066
2. APIRouter (Kiến trúc phân tầng)
Thay thế cho: Blueprint (trong thư mục views/).
Nhiệm vụ: Chia nhỏ dự án HR thành các mảng: Nhân viên, Phòng ban, Phân quyền.
Điểm mới: Không chỉ gom nhóm URL, APIRouter còn tự động gom nhóm Tài liệu (Docs).
Trong Flask, Blueprint chỉ để quản lý code.
Trong FastAPI, APIRouter giúp trang /docs hiển thị ngăn nắp: Mục "Employee", Mục "Department" riêng biệt với đầy đủ hướng dẫn sử dụng.
3. Pydantic (Người gác cổng dữ liệu)
Thay thế cho: Các câu lệnh if/else kiểm tra dữ liệu (Validation).
Nhiệm vụ: Định nghĩa "cái khuôn" cho dữ liệu.
Điểm mới: Đây là thứ Flask không có sẵn.
Bạn khai báo: age: int. Nếu người dùng nhập "ba mươi", Pydantic tự chặn lại và báo lỗi ngay lập tức, không để dữ liệu rác chạm vào câu lệnh SQL pyodbc của bạn.
Nó giúp chuyển kết quả từ cursor.fetchall() sang định dạng JSON cực đẹp để trả về cho file .js ở giao diện.
Tổng kết sự thay đổi trong dự án SMILE HR của bạn:
Thành phần	Ở Flask (Cũ)	Ở FastAPI (Mới)		Lợi ích cho SMILE HR
Server		Waitress	Uvicorn			Web chạy cực nhanh, không bị treo khi đông người.
Tổ chức		Blueprint	APIRouter		Code sạch, tự động có trang hướng dẫn API (/docs).
Dữ liệu	      if/else thủ công	Pydantic		Chặn đứng dữ liệu sai trước khi lưu vào SQL Server.
